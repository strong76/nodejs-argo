discord_translator.py

import discord
from discord.ext import commands
import googletrans
from googletrans import Translator
import asyncio
import json
import os

# Load configuration
with open('config.json', 'r') as f:
    config = json.load(f)

# Initialize translator
translator = Translator()

# Set bot prefix and intents
intents = discord.Intents.default()
intents.message_content = True
bot = commands.Bot(command_prefix='!', intents=intents)

class TranslationBot(commands.Cog):
    def __init__(self, bot):
        self.bot = bot
        self.auto_channels = config.get('auto_channels', [])
        self.default_lang = config.get('default_lang', 'en')
        
    @commands.Cog.listener()
    async def on_ready(self):
        print(f'{self.bot.user} has logged in successfully!')
        print(f'Default translation language: {self.default_lang}')
        print(f'Auto-translation channels: {self.auto_channels}')
        
    @commands.Cog.listener()
    async def on_message(self, message):
        # Ignore bot's own messages
        if message.author == self.bot.user:
            return
            
        # Check if in auto-translation channels
        if str(message.channel.id) in self.auto_channels and not message.content.startswith('!'):
            # Detect message language
            detected = translator.detect(message.content)
            src_lang = detected.lang
            
            # Translate if message language is not default language
            if src_lang != self.default_lang:
                try:
                    translation = translator.translate(
                        message.content, 
                        src=src_lang, 
                        dest=self.default_lang
                    )
                    
                    # Create embed message to show translation
                    embed = discord.Embed(
                        title="Translation Result",
                        color=discord.Color.blue()
                    )
                    embed.add_field(name="Original", value=message.content, inline=False)
                    embed.add_field(name="Translation", value=translation.text, inline=False)
                    embed.set_footer(text=f"{src_lang} → {self.default_lang}")
                    
                    await message.channel.send(embed=embed)
                except Exception as e:
                    print(f"Translation error: {e}")
                    
    @commands.command(name="translate", aliases=["tr"])
    async def translate_command(self, ctx, *, text):
        """Translate text to default language"""
        try:
            translation = translator.translate(text, dest=self.default_lang)
            
            embed = discord.Embed(
                title="Translation Result",
                color=discord.Color.green()
            )
            embed.add_field(name="Original", value=text, inline=False)
            embed.add_field(name="Translation", value=translation.text, inline=False)
            embed.set_footer(text=f"{translation.src} → {self.default_lang}")
            
            await ctx.send(embed=embed)
        except Exception as e:
            await ctx.send(f"Translation error: {e}")
            
    @commands.command(name="setlang")
    @commands.has_permissions(administrator=True)
    async def set_language(self, ctx, lang_code):
        """Set default translation language"""
        supported_langs = googletrans.LANGUAGES
        
        if lang_code in supported_langs:
            self.default_lang = lang_code
            config['default_lang'] = lang_code
            
            # Save to configuration file
            with open('config.json', 'w') as f:
                json.dump(config, f, indent=4)
                
            await ctx.send(f"Default translation language set to: {supported_langs[lang_code]} ({lang_code})")
        else:
            await ctx.send(f"Unsupported language code. Please use valid language codes like: en, zh, ja, ko, etc.")
            
    @commands.command(name="autotranslate")
    @commands.has_permissions(administrator=True)
    async def toggle_auto_translate(self, ctx):
        """Enable/disable auto-translation in current channel"""
        channel_id = str(ctx.channel.id)
        
        if channel_id in self.auto_channels:
            self.auto_channels.remove(channel_id)
            status = "Disabled"
        else:
            self.auto_channels.append(channel_id)
            status = "Enabled"
            
        config['auto_channels'] = self.auto_channels
        
        with open('config.json', 'w') as f:
            json.dump(config, f, indent=4)
            
        await ctx.send(f"Auto-translation {status} in this channel.")

async def main():
    async with bot:
        await bot.add_cog(TranslationBot(bot))
        await bot.start(config['discord_token'])

if __name__ == "__main__":
    asyncio.run(main())
